function pulseStroke(stroke, colorA, colorB)
	task.spawn(function()
		while stroke and stroke.Parent do
			tween(stroke, TweenInfo.new(1.2, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut), {
				Color = colorB,
				Transparency = 0.1,
			}):Play()
			task.wait(1.2)
			if not stroke.Parent then
				break
			end
			tween(stroke, TweenInfo.new(1.2, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut), {
				Color = colorA,
				Transparency = 0.45,
			}):Play()
			task.wait(1.2)
		end
	end)
end

function wireButtonAnim(btn)
	local scale = btn:FindFirstChildOfClass("UIScale")
	if not scale then
		scale = Instance.new("UIScale")
		scale.Scale = 1
		scale.Parent = btn
	end
	local base = scale.Scale

	btn.MouseEnter:Connect(function()
		tween(scale, TweenInfo.new(0.15, Enum.EasingStyle.Back, Enum.EasingDirection.Out), { Scale = base * 1.06 }):Play()
	end)
	btn.MouseLeave:Connect(function()
		tween(scale, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), { Scale = base }):Play()
	end)
	btn.MouseButton1Down:Connect(function()
		tween(scale, TweenInfo.new(0.08, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), { Scale = base * 0.94 }):Play()
	end)
	btn.MouseButton1Up:Connect(function()
		tween(scale, TweenInfo.new(0.12, Enum.EasingStyle.Back, Enum.EasingDirection.Out), { Scale = base * 1.04 }):Play()
		task.delay(0.12, function()
			if scale.Parent then
				tween(scale, TweenInfo.new(0.1, Enum.EasingStyle.Quad), { Scale = base }):Play()
			end
		end)
	end)
	table.insert(UI.animatedButtons, btn)
end

function flashLabel(label, color)
	if not label then
		return
	end
	local original = label.TextColor3
	label.TextColor3 = color
	tween(label, TweenInfo.new(0.35, Enum.EasingStyle.Quad), { TextTransparency = 0.2 }):Play()
	task.delay(0.35, function()
		if label.Parent then
			tween(label, TweenInfo.new(0.35, Enum.EasingStyle.Quad), { TextTransparency = 0 }):Play()
			label.TextColor3 = original
		end
	end)
end

function getTableFromSeat(seat)
	if not (seat and seat:IsA("Seat")) then
		return nil, nil
	end
	local chair = seat.Parent
	local tableModel = chair and chair.Parent
	local tablesFolder = getTablesFolder()
	if tablesFolder and tableModel and tableModel.Parent == tablesFolder then
		return tableModel, chair.Name
	end
	return nil, nil
end

getPlayerTableStatus = function(player)
	local character = player.Character
	local humanoid = character and character:FindFirstChildOfClass("Humanoid")
	local seat = humanoid and humanoid.SeatPart
	if not seat then
		return false, nil, nil
	end
	local tableModel, chairName = getTableFromSeat(seat)
	if tableModel then
		return true, tableModel.Name, chairName
	end
	return false, nil, nil
end

isLocalPlayerAtTable = function(tableName)
	local seated, myTable = getPlayerTableStatus(LocalPlayer)
	if not seated then
		return false
	end
	if tableName and myTable ~= tableName then
		return false
	end
	return true
end

function getOtherSeat(tableModel, occupiedChair)
	for _, chairName in ipairs({ "Chair1", "Chair2" }) do
		if chairName ~= occupiedChair then
			local chair = tableModel:FindFirstChild(chairName)
			local seat = chair and chair:FindFirstChildWhichIsA("Seat", true)
			if seat and seat.Occupant == nil then
				return seat, chairName
			end
		end
	end
end

function getSeatForChair(tableModel, chairName)
	if not tableModel or not chairName then
		return nil
	end
	local chair = tableModel:FindFirstChild(chairName)
	return chair and chair:FindFirstChildWhichIsA("Seat", true)
end

function confirmTableEnter(tableName, chairName)
	if not tableName or not chairName then
		return
	end
	pcall(function()
		REMOTES.EnterRequest:FireServer(tableName, chairName, true)
	end)
	if REMOTES.EquipBrainrot then
		pcall(function()
			REMOTES.EquipBrainrot:FireServer(nil)
		end)
	end
end

function waitForTableJoin(tableName, chairName, timeout)
	local deadline = tick() + (timeout or 0.5)
	while tick() < deadline do
		local seated, tName, cName = getPlayerTableStatus(LocalPlayer)
		if seated and tName == tableName and (not chairName or cName == chairName) then
			return true
		end
		local character = LocalPlayer.Character
		local humanoid = character and character:FindFirstChildOfClass("Humanoid")
		local seatPart = humanoid and humanoid.SeatPart
		if seatPart then
			local tableModel, seatChair = getTableFromSeat(seatPart)
			if tableModel and tableModel.Name == tableName then
				confirmTableEnter(tableName, seatChair or chairName)
				return true
			end
		end
		task.wait(0.05)
	end
	return false
end

function requestTableEnter(tableName, chairName, attempts, waitSec)
	attempts = attempts or 4
	waitSec = waitSec or 0.4
	for _ = 1, attempts do
		confirmTableEnter(tableName, chairName)
		if waitForTableJoin(tableName, chairName, waitSec) then
			return true
		end
		task.wait(0.08)
	end
	return false
end

function refreshLuckEventMult()
	local now = os.time()
	local best = 1
	for boostName, expiry in pairs(State.luckBoosts or {}) do
		if now < expiry and LUCK_BOOST_MULT[boostName] and LUCK_BOOST_MULT[boostName] > best then
			best = LUCK_BOOST_MULT[boostName]
		end
	end
	State.luckEventMult = best
end

	local stored = State.playerLuck[player.UserId]
	if stored ~= nil then
		return math.floor(stored)
	end
	local displayed = readBillboardLuck(player)
	if displayed and displayed > 0 then
		local mult = State.luckEventMult or 1
		if mult > 1 then
			return math.max(0, math.floor(displayed / mult + 0.5))
		end
		return math.floor(displayed)
	end
	return 0
end

function getDisplayLuck(player)
	local displayed = readBillboardLuck(player)
	if displayed and displayed > 0 then

	end
	local base = getBaseLuck(player)
	if player == LocalPlayer then
		return math.floor(base * (State.luckEventMult or 1))

	return base
end

function formatMoney(amount)
	amount = math.floor(tonumber(amount) or 0)
	if amount >= 1e12 then
		return string.format("$%.2fT", amount / 1e12)
	end
	if amount >= 1e9 then
		return string.format("$%.2fB", amount / 1e9)
	end
	if amount >= 1e6 then
		return string.format("$%.2fM", amount / 1e6)
	end
	if amount >= 1e3 then
		return string.format("$%.2fK", amount / 1e3)
	end
	return "$" .. tostring(amount)
end

function parseMoneyThreshold(text)
	text = string.gsub(string.lower(tostring(text or "")), "%s+", "")
	if text == "" then
		return DEFAULT_SELL_THRESHOLD
	end
	local num, suffix = string.match(text, "^([%d%.]+)([kmbt]?)$")
	if not num then
		local plain = string.match(text, "^(%d+)$")
		if plain then
			return math.floor(tonumber(plain))
		end
		return State.moneySellThreshold or DEFAULT_SELL_THRESHOLD
	end
	local mult = 1
	if suffix == "k" then
		mult = 1e3
	elseif suffix == "m" then
		mult = 1e6
	elseif suffix == "b" then
		mult = 1e9
	elseif suffix == "t" then
		mult = 1e12
	end
	return math.floor((tonumber(num) or 0) * mult)
end

function formatDPS(amount)
	amount = math.floor(tonumber(amount) or 0)
	if amount >= 1e12 then
		return formatMoney(amount) .. "/s"
	end
	if amount >= 1e9 then
		return string.format("$%.1fB/s", amount / 1e9)
	end
	if amount >= 1e6 then
		return string.format("$%.1fM/s", amount / 1e6)
	end
	if amount >= 1e3 then
		return string.format("$%.1fK/s", amount / 1e3)
	end
	return "$" .. tostring(amount) .. "/s"
end

function getBrainrotDPS(item)
	local info = BrainrotInfo[item.name]
	if not info then
		return 0
	end
	return math.floor(
		info.BaseValue
			* (VARIANT_MULT[item.variant or "Normal"] or 1)
			* 0.1
			* (State.cashMult or 1)
			* (State.vipDouble and 2 or 1)
	)
end

function getBrainrotRollDPS(name, variant)
	local info = BrainrotInfo[name]
	if not info then
		return 0
	end
	-- Table roll UI uses BaseValue * variant * 0.1 only (DecidingScript — no rebirth cash mult).
	return math.floor(info.BaseValue * (VARIANT_MULT[variant or "Normal"] or 1) * 0.1)
end

function getDiscordThumbnailUrl(image, size)
	local assetId = extractRobloxAssetId(image)
	if not assetId then
		return nil
	end
	size = size or "512x512"
	local apiUrl = "https://thumbnails.roblox.com/v1/assets?assetIds="
		.. assetId
		.. "&size="
		.. size
		.. "&format=Png&isCircular=false"
	local response = httpRequest({ Url = apiUrl, Method = "GET" })
	if response and response.Body then
		local ok, data = pcall(function()
			return HttpService:JSONDecode(response.Body)
		end)
		if ok and data and type(data.data) == "table" and data.data[1] then
			local entry = data.data[1]
			if entry.state == "Completed" and entry.imageUrl and entry.imageUrl ~= "" then
				return entry.imageUrl
			end
		end
	end
	return "https://www.roblox.com/asset-thumbnail/image?assetId="
		.. assetId
		.. "&width=512&height=512&format=png"
end

function getTopBrainrots(count)
	local items = {}
	for _, item in ipairs(State.inventory) do
		if item.name and item.name ~= "" then
			table.insert(items, item)
		end
	end
	table.sort(items, function(a, b)
		local rankA = RARITY_RANK[a.rarity] or 99
		local rankB = RARITY_RANK[b.rarity] or 99
		if rankA ~= rankB then
			return rankA < rankB
		end
		return getBrainrotDPS(b) < getBrainrotDPS(a)
	end)
	local top = {}
	for i = 1, math.min(count or 3, #items) do
		top[i] = items[i]
	end
	return top
end

function normalizeOwnedChairs(owned)
	local list = {}
	if type(owned) ~= "table" then
		return list
	end
	if #owned > 0 then
		for _, v in ipairs(owned) do
			if type(v) == "string" and v ~= "" then
				table.insert(list, v)
			end
		end
	else
		for k, v in pairs(owned) do
			if type(k) == "string" and k ~= "" and (v == true or v == 1) then
				table.insert(list, k)
			elseif type(v) == "string" and v ~= "" then
				table.insert(list, v)
			end
		end
	end
	return list
end

function showAnimatedBottomNotify(message, kind, duration, options)
	duration = duration or 3.5
	options = options or {}
	if not UI.notifyFrame or not UI.notifyLabel then
		return
	end
	UI.notifyHideToken = (UI.notifyHideToken or 0) + 1
	local token = UI.notifyHideToken
	local textColor, bgColor, strokeColor = getNotifyStyle(kind)
	local textSize = options.textSize or 13
	local maxWidth = options.maxWidth or 400

	UI.notifyLabel.Text = message
	UI.notifyLabel.TextSize = textSize
	UI.notifyLabel.TextColor3 = textColor
	UI.notifyLabel.TextXAlignment = Enum.TextXAlignment.Left
	UI.notifyLabel.TextYAlignment = Enum.TextYAlignment.Top
	UI.notifyFrame.BackgroundColor3 = bgColor
	if UI.notifyStroke then
		UI.notifyStroke.Color = strokeColor
	end

	local textService = game:GetService("TextService")
	local textSizeInfo = textService:GetTextSize(message, textSize, UI.notifyLabel.Font, Vector2.new(maxWidth - 24, 200))
	local width = math.clamp(textSizeInfo.X + 28, 300, maxWidth)
	local height = math.max(78, textSizeInfo.Y + 24)
	UI.notifyFrame.Size = UDim2.new(0, width, 0, height)
	UI.notifyFrame.Visible = true
	UI.notifyFrame.BackgroundTransparency = 0.08

	if UI.notifyScale then
		UI.notifyScale.Scale = 0.82
	end
	UI.notifyFrame.Position = UDim2.new(1, 36, 1, -16)

	local slideIn = TweenInfo.new(0.42, Enum.EasingStyle.Back, Enum.EasingDirection.Out)
	tween(UI.notifyFrame, slideIn, {
		Position = UDim2.new(1, -16, 1, -16),
	}):Play()
	if UI.notifyScale then
		local targetScale = options.scale or 1.08
		tween(UI.notifyScale, slideIn, { Scale = targetScale }):Play()
	end
	if UI.notifyStroke then
		tween(UI.notifyStroke, TweenInfo.new(0.35), { Transparency = 0.15 }):Play()
	end

	task.delay(duration, function()
		if token ~= UI.notifyHideToken or not UI.notifyFrame then
			return
		end
		local slideOut = TweenInfo.new(0.32, Enum.EasingStyle.Quad, Enum.EasingDirection.In)
		local tw = tween(UI.notifyFrame, slideOut, {
			Position = UDim2.new(1, 40, 1, -16),
			BackgroundTransparency = 1,
		})
		if UI.notifyScale then
			tween(UI.notifyScale, slideOut, { Scale = 0.88 }):Play()
		end
		tw:Play()
		tw.Completed:Connect(function()
			if token == UI.notifyHideToken and UI.notifyFrame then
				UI.notifyFrame.Visible = false
				UI.notifyFrame.BackgroundTransparency = 0.08
			end
		end)
	end)
end

	if not State.autoIndexReward then
		return false
	end
	local now = tick()
	if now - (State.lastIndexSync or 0) >= INDEX_SYNC_INTERVAL then
		State.lastIndexSync = now
		syncIndexFromGame()
	end
	local pending = State.indexSlotsPending
	if pending <= 0 then
		pending = _G.IndexSlotsPending or 0
	end
	if pending <= 0 or not REMOTES.ClaimIndexSlots then
		return false
	end
	if now - (State.lastIndexClaim or 0) < INDEX_CLAIM_COOLDOWN then
		return false
	end
	State.lastIndexClaim = now
	REMOTES.ClaimIndexSlots:FireServer()
	showAnimatedBottomNotify("Claimed +10 inventory slots!", "index", 3)
	setStatus("Index reward claimed (+10 slots)")
	return true
end

function computeMerchantSlotStock(restockId, slotIndex)
	local totalWeight = 0
	for _, entry in ipairs(MERCHANT_STOCK_WEIGHTS) do
		totalWeight += entry.weight
	end
	local seed = ((restockId * 2654435761 + slotIndex * 2246822519) % 4294967296 + 1831565813) % 4294967296
	local mixed = (seed + seed * 32 % 4294967296) % 4294967296 * 3526669397 % 4294967296
	local folded = (mixed + math.floor(mixed / 2048)) % 4294967296
	local rolled = (folded + folded * 32 % 4294967296) % 4294967296 * 1761867953 % 4294967296 / 4294967296 * totalWeight
	local cumulative = 0
	for _, entry in ipairs(MERCHANT_STOCK_WEIGHTS) do
		cumulative += entry.weight
		if rolled <= cumulative then
			return entry.stock
		end
	end
	return 0
end

	State.merchantRestockId = restockId or 0
	State.merchantPurchasedThisCycle = {}
	for index, slot in ipairs(MERCHANT_SLOTS) do
		State.merchantStock[slot.id] = computeMerchantSlotStock(restockId or 0, index)
	end
end

function getMerchantAvailableStock(slotId)
	local base = State.merchantStock[slotId] or 0
	local bought = State.merchantPurchasedThisCycle[slotId] or 0
	return math.max(0, base - bought)
end

function tryMerchantAutoBuy()
	if not State.autoBuyMerchant or not REMOTES.MerchantBuy then
		return false
	end
	local now = tick()
	if now - State.lastMerchantBuy < MERCHANT_BUY_COOLDOWN then
		return false
	end
	if State.merchantPendingBuy and next(State.merchantPendingBuy) then
		return false
	end
	local money = getLocalMoney()
	for index, slot in ipairs(MERCHANT_SLOTS) do
		if State.merchantSlotsSelected[slot.id] then
			if getMerchantAvailableStock(slot.id) <= 0 then
				continue
			end
			if money < slot.price then
				continue
			end
			State.merchantPendingBuy[slot.id] = true
			State.lastMerchantBuy = now
			REMOTES.MerchantBuy:FireServer(slot.id)
			setStatus("Merchant buy -> " .. slot.label)
			return true
		end
	end
	return false
end

	return State.chairTraits[chairName]
end

function shouldStopTraitReroll(traitName)
	if not traitName then
		return false
	end
	return State.stopTraits[traitName] == true
end

function tryAutoTraitReroll()
	if not State.autoTraitReroll or not REMOTES.RequestTraitReroll then
		return false
	end
	local chairName = State.traitRerollChair
	if not chairName or chairName == "" then
		return false
	end
	if not ownsChair(chairName) then
		return false
	end
	if (State.traitRerolls or 0) < 1 then
		return false
	end
	local now = tick()
	if now - State.lastTraitReroll < TRAIT_REROLL_COOLDOWN then
		return false
	end
	local currentTrait = getChairTrait(chairName)
	if shouldStopTraitReroll(currentTrait) then
		return false
	end
	State.traitRerollPendingChair = chairName
	State.traitRerollPendingTrait = currentTrait
	State.lastTraitReroll = now
	notifyTraitRollStart(chairName)
	REMOTES.RequestTraitReroll:FireServer(chairName)
	setStatus("Trait reroll -> " .. getChairDisplayName(chairName))
	return true
end

function handleChairTraitUpdated(chairName, traitName)
	if chairName then
		State.chairTraits[chairName] = traitName
	end
	updateTraitTabLabels()

	local isTrackedReroll = State.traitRerollPendingChair == chairName
		or (State.autoTraitReroll and chairName == State.traitRerollChair)

	if isTrackedReroll then
		local stopped = shouldStopTraitReroll(traitName)
		notifyTraitRollResult(chairName, traitName, stopped)
		tryWebhookOnTraitReroll(chairName, traitName)
		if State.traitRerollPendingChair == chairName then
			State.traitRerollPendingChair = nil
			State.traitRerollPendingTrait = nil
		end
	end
end

getBestOwnedChair = function()
	local best = "DefaultChair"
	local bestLuck = getChairLuck("DefaultChair")
	local candidates = { "DefaultChair", "VIPChair" }
	for _, chairName in ipairs(CHAIR_ORDER) do
		table.insert(candidates, chairName)
	end
	for _, chairName in ipairs(SPECIAL_CHAIR_ORDER) do
		table.insert(candidates, chairName)
	end
	for _, chairName in ipairs(State.specialOwnedChairs or {}) do
		table.insert(candidates, chairName)
	end
	for _, chairName in ipairs(State.ownedChairs or {}) do
		table.insert(candidates, chairName)
	end
	local seen = {}
	for _, chairName in ipairs(candidates) do
		if not seen[chairName] and ownsChair(chairName) then
			seen[chairName] = true
			local luck = getChairLuck(chairName)
			if luck > bestLuck then
				bestLuck = luck
				best = chairName
			end
		end
	end
	return best
end

function httpRequest(options)
	local req = syn and syn.request or http and http.request or fluxus and fluxus.request or request
	if req then
		return req(options)
	end
	if HttpService.HttpEnabled then
		local ok, result = pcall(function()
			if options.Method == "POST" then
				return HttpService:PostAsync(options.Url, options.Body, Enum.HttpContentType.ApplicationJson)
			end
			return HttpService:GetAsync(options.Url)
		end)
		if ok then
			return { Success = true, Body = result }
		end
		return { Success = false, Body = tostring(result) }
	end
	return nil
end

function isHttpSuccess(response)
	if not response then
		return false
	end
	if response.Success == true then
		return true
	end
	local code = response.StatusCode or response.status or response.Status
	return code == 200 or code == 204
end

function rarityToEmbedColor(rarity)
	local c = RARITY_COLORS[rarity]
	if not c then
		return 5814783
	end
	return math.floor(c.R * 255) * 65536 + math.floor(c.G * 255) * 256 + math.floor(c.B * 255)
end

	elseif inst:IsA("ParticleEmitter") or inst:IsA("Trail") or inst:IsA("Beam") or inst:IsA("Fire") or inst:IsA("Smoke") or inst:IsA("Sparkles") then
		inst.Enabled = false
	elseif inst:IsA("BasePart") then
		inst.Material = Enum.Material.SmoothPlastic
		inst.CastShadow = false
		inst.Reflectance = 0
	elseif inst:IsA("MeshPart") then
		inst.TextureID = ""
		inst.CastShadow = false
	elseif inst:IsA("SpecialMesh") then
		inst.TextureId = ""
	end
end

function trackHidePlayer(player)
	if player == LocalPlayer then
		return
	end
	local function onChar(char)
		if State.hideOtherPlayers then
			hideCharacter(char)
		end
		table.insert(
			State.hidePlayerConns,
			char.DescendantAdded:Connect(function(inst)
				if State.hideOtherPlayers then
					if inst:IsA("BasePart") then
						inst.LocalTransparencyModifier = 1
					elseif inst:IsA("Decal") or inst:IsA("Texture") then
						inst.Transparency = 1
					end
				end
			end)
		)
	end
	if player.Character then
		onChar(player.Character)
	end
	table.insert(
		State.hidePlayerConns,
		player.CharacterAdded:Connect(onChar)
	)
end

function applyHideOtherPlayers(enabled)
	State.hideOtherPlayers = enabled
	for _, conn in ipairs(State.hidePlayerConns) do
		conn:Disconnect()
	end
	table.clear(State.hidePlayerConns)
	if enabled then
		for _, player in ipairs(Players:GetPlayers()) do
			trackHidePlayer(player)
		end
		table.insert(
			State.hidePlayerConns,
			Players.PlayerAdded:Connect(function(player)
				trackHidePlayer(player)
			end)
		)
	else
		for _, player in ipairs(Players:GetPlayers()) do
			if player ~= LocalPlayer and player.Character then
				showCharacter(player.Character)
			end
		end
	end
	if UI.hidePlayersToggle then
		UI.hidePlayersToggle.set(enabled, true)
	end
end

	local vp = getViewportSize()

function getHubUiScale()
	local vp = getViewportSize()
	local shortSide = math.min(vp.X, vp.Y)
	local touchPrimary = UserInputService.TouchEnabled and not UserInputService.KeyboardEnabled
	local compact = touchPrimary or shortSide <= 720 or isDeltaExecutor()

	if compact then
		-- Mobile / LDPlayer / Delta: readable size, not tiny
		return math.clamp(shortSide / 560, 0.82, 1.05)
	end
	if shortSide <= 900 then
		return math.clamp(shortSide / 780, 0.86, 0.98)
	end
	return 0.94
end

function clampHubDragPosition(pos)
	if not UI.hubRoot then
		return pos
	end
	local width = UI.hubRoot.Size.X.Offset
	local height = UI.hubRoot.Size.Y.Offset
	local vpX, vpY = getHubGuiBounds()
	local margin = 6
	local x = math.clamp(pos.X.Offset, margin, math.max(margin, vpX - width - margin))
	local minY = margin
	local maxY = vpY - height - margin
	if maxY < minY then
		minY = vpY - height
		maxY = margin
	end
	local y = math.clamp(pos.Y.Offset, math.min(minY, maxY), math.max(minY, maxY))
	return UDim2.new(0, x, 0, y)
end

	local size = getMiniFabSize()
	local vp = getViewportSize()

	local y = math.clamp(pos.Y.Offset, -(vp.Y - size - 4), -4)
	return UDim2.new(0, x, 1, y)
end

function wireDraggable(target, handles, config)
	config = config or {}
	local dragging = false
	local dragInput = nil
	local dragStart = nil
	local startPos = nil
	local moved = false

	local function inputPos(input)
		return Vector2.new(input.Position.X, input.Position.Y)
	end

	local function onChanged(input)
		if not dragging then
			return
		end
		local inputType = input.UserInputType
		if inputType ~= Enum.UserInputType.MouseMovement and inputType ~= Enum.UserInputType.Touch then
			return
		end
		if dragInput and inputType == Enum.UserInputType.Touch and input ~= dragInput then
			return
		end
		local delta = inputPos(input) - dragStart
		if delta.Magnitude > DRAG_CLICK_THRESHOLD then
			moved = true
		end
		local newPos = UDim2.new(
			startPos.X.Scale,
			startPos.X.Offset + delta.X,
			startPos.Y.Scale,
			startPos.Y.Offset + delta.Y
		)
		if config.clamp then
			newPos = config.clamp(newPos)
		end
		target.Position = newPos
	end

	local function onEnded(input)
		if not dragging then
			return
		end
		if input.UserInputType ~= Enum.UserInputType.MouseButton1
			and input.UserInputType ~= Enum.UserInputType.Touch then
			return
		end
		if dragInput and input ~= dragInput then
			return
		end
		dragging = false
		dragInput = nil
		if moved and config.onDragEnd then
			config.onDragEnd(target.Position, moved)
		elseif not moved and config.onClick then
			config.onClick()
		end
		moved = false
	end

	for _, handle in ipairs(handles) do
		if handle then
			handle.Active = true
			if handle:IsA("GuiButton") then
				handle.AutoButtonColor = false
			end
			handle.InputBegan:Connect(function(input, gameProcessed)
				if gameProcessed then
					return
				end
				if input.UserInputType ~= Enum.UserInputType.MouseButton1
					and input.UserInputType ~= Enum.UserInputType.Touch then
					return
				end
				if config.canStart and not config.canStart() then
					return
				end
				dragging = true
				moved = false
				dragInput = input
				dragStart = inputPos(input)
				startPos = target.Position
				if config.onDragStart then
					config.onDragStart()
				end
			end)
			handle.InputChanged:Connect(function(input)
				if dragging and input == dragInput then
					onChanged(input)
				end
			end)
		end
	end

	UserInputService.InputChanged:Connect(onChanged)
	UserInputService.InputEnded:Connect(onEnded)
end

function tryAutoRebirth()
	if not State.autoRebirth or not REMOTES.RebirthRequest then
		return false
	end
	if (State.rebirthCount or 0) >= MAX_REBIRTH then
		return false
	end
	local cost = State.rebirthCost or getRebirthCost(State.rebirthCount)
	if not cost or cost <= 0 then
		return false
	end
	local money = getLocalMoney()
	if money < cost then
		return false
	end
	local now = tick()
	if now - (State.lastRebirth or 0) < REBIRTH_COOLDOWN then
		return false
	end
	State.lastRebirth = now
	REMOTES.RebirthRequest:FireServer()
	setStatus("Auto rebirth -> " .. formatMoney(cost))
	return true
end

function shouldForceLeaveSeat()
	if not isLocalSeated() or State.localRollActive then
		return false
	end
	if isDecisionOpen and isDecisionOpen() then
		return false
	end
	local opponent = getTableOpponent()
	if not opponent then
		if State.autoJoinRandom or State.autoJoinRandomChair then
			return false
		end
		return true
	end
	if State.autoJoin and State.selectedPlayer and opponent == State.selectedPlayer then
		return false
	end
	if State.autoJoinHighestLuck then
		local top = getHighestLuckEntry()
		if top and opponent == top.player then
			return false
		end
	end
	if State.autoJoinMinLuck and getLuck(opponent) >= MIN_LUCK_JOIN then
		return false
	end
	if State.autoJoinRandom or State.autoJoinRandomChair then
		return false
	end
	return true
end

function hasAnyAutoJoinMode()
	return State.autoJoin
		or State.autoJoinHighestLuck
		or State.autoJoinMinLuck
		or State.autoJoinRandom
		or State.autoJoinRandomChair
end

function canJoinPlayer(target)
	if not target or target == LocalPlayer then
		return false
	end
	local atTable, _, chairName = getPlayerTableStatus(target)
	if not atTable then
		return false
	end
	local character = target.Character
	local humanoid = character and character:FindFirstChildOfClass("Humanoid")
	local targetSeat = humanoid and humanoid.SeatPart
	local tableModel = getTableFromSeat(targetSeat)
	if not tableModel then
		return false
	end
	local openSeat = getOtherSeat(tableModel, chairName)
	return openSeat ~= nil
end

function getPlayersSortedByLuck()
	local list = {}
	for _, player in ipairs(Players:GetPlayers()) do
		if player ~= LocalPlayer then
			table.insert(list, {
				player = player,
				luck = getLuck(player),
			})
		end
	end
	table.sort(list, function(a, b)
		return a.luck > b.luck
	end)
	return list
end

function getHighestLuckEntry()
	local sorted = getPlayersSortedByLuck()
	return sorted[1]
end

function findHighestLuckJoinTarget()
	local top = getHighestLuckEntry()
	if not top then
		return nil, 0, 0, "none"
	end

	if getPlayerTableStatus(top.player) then
		if canJoinPlayer(top.player) then
			return top.player, top.luck, 1, "join"
		end
		return nil, top.luck, 1, "wait"
	end

	return nil, top.luck, 1, "wait"
end

function getPlayersWithMinLuck(minLuck)
	local list = {}
	for _, player in ipairs(Players:GetPlayers()) do
		if player ~= LocalPlayer then
			local luck = getLuck(player)
			if luck >= minLuck then
				table.insert(list, {
					player = player,
					luck = luck,
				})
			end
		end
	end
	table.sort(list, function(a, b)
		return a.luck > b.luck
	end)
	return list
end

function findMinLuckJoinTarget()
	for _, entry in ipairs(getPlayersWithMinLuck(MIN_LUCK_JOIN)) do
		if getPlayerTableStatus(entry.player) and canJoinPlayer(entry.player) then
			return entry.player, entry.luck, 1, "join"
		end
	end
	return nil, 0, 0, "none"
end

function findRandomJoinTarget()
	local list = {}
	for _, player in ipairs(Players:GetPlayers()) do
		if player ~= LocalPlayer and getPlayerTableStatus(player) and canJoinPlayer(player) then
			table.insert(list, {
				player = player,
				luck = getLuck(player),
			})
		end
	end
	if #list == 0 then
		return nil, 0, "none"
	end
	local pick = list[math.random(1, #list)]
	return pick.player, pick.luck, "join"
end

function tryJoinWithRetries(joinFn)
	for attempt = 1, JOIN_MAX_RETRIES do
		local ok, msg = joinFn()
		if ok or isLocalSeated() then
			return true, msg or "joined"
		end
		if attempt < JOIN_MAX_RETRIES then
			task.wait(0.2 + attempt * 0.08)
		end
	end
	return false, "failed to sit after retries"
end

function getTableSeatInfo(tableModel)
	local chair1 = tableModel:FindFirstChild("Chair1")
	local chair2 = tableModel:FindFirstChild("Chair2")
	if not (chair1 and chair2) then
		return nil
	end
	local seat1 = chair1:FindFirstChildOfClass("Seat")
	local seat2 = chair2:FindFirstChildOfClass("Seat")
	if not (seat1 and seat2) then
		return nil
	end
	local occ1 = seat1.Occupant ~= nil
	local occ2 = seat2.Occupant ~= nil
	return {
		tableModel = tableModel,
		tableName = tableModel.Name,
		seat1 = seat1,
		seat2 = seat2,
		chair1Name = "Chair1",
		chair2Name = "Chair2",
		occ1 = occ1,
		occ2 = occ2,
		occupiedCount = (occ1 and 1 or 0) + (occ2 and 1 or 0),
	}
end

function getJoinableRandomTableSeats()
	local tablesFolder = getTablesFolder()
	if not tablesFolder then
		return {}
	end
	local list = {}
	for _, tableModel in ipairs(tablesFolder:GetChildren()) do
		local info = getTableSeatInfo(tableModel)
		if info and info.occupiedCount < 2 then
		if info.occupiedCount == 1 then
			local seat, chairName, waitingPlayer
			if not info.occ1 then
				seat = info.seat1
				chairName = info.chair1Name
				waitingPlayer = getPlayerFromSeatOccupant(info.seat2)
			else
				seat = info.seat2
				chairName = info.chair2Name
				waitingPlayer = getPlayerFromSeatOccupant(info.seat1)
			end
			table.insert(list, {
				seat = seat,
				tableName = info.tableName,
				chairName = chairName,
				waitingPlayer = waitingPlayer,
				kind = "waiting",
			})
		else
			local pickFirst = math.random() < 0.5
			table.insert(list, {
				seat = pickFirst and info.seat1 or info.seat2,
				tableName = info.tableName,
				chairName = pickFirst and info.chair1Name or info.chair2Name,
				waitingPlayer = nil,
				kind = "empty",
			})
		end
		end
	end
	return list
end

function findRandomTableSeat()
	local candidates = getJoinableRandomTableSeats()
	if #candidates == 0 then
		return nil
	end
	return candidates[math.random(1, #candidates)]
end

function leaveTable(force)
	local character = LocalPlayer.Character
	local humanoid = character and character:FindFirstChildOfClass("Humanoid")
	local root = character and character:FindFirstChild("HumanoidRootPart")
	if not humanoid or not humanoid.SeatPart then
		return true
	end

	local seat = humanoid.SeatPart
	local tableModel, chairName = getTableFromSeat(seat)
	if tableModel and chairName then
		REMOTES.EnterRequest:FireServer(tableModel.Name, chairName, false)
	end

	local jumpAttempts = force and 14 or 4
	for _ = 1, jumpAttempts do
		if not humanoid.SeatPart then
			break
		end
		pcall(function()
			humanoid.Sit = false
		end)
		humanoid.Jump = true
		task.wait(force and 0.05 or 0.08)
	end

	if force and humanoid.SeatPart and root then
		for _ = 1, 14 do
			if not humanoid.SeatPart then
				break
			end
			standOnLeavePlatform(humanoid, root)
			task.wait(0.06)
		end
		if not humanoid.SeatPart and root then
			local plat = getLeavePlatform()
			local standY = plat.Position.Y + (plat.Size.Y / 2) + 3
			root.CFrame = CFrame.new(plat.Position.X, standY, plat.Position.Z)
		end
	end

	return humanoid.SeatPart == nil
end

function getLuckUpgradeCost(amount, baseLuck)
	local level = math.floor((baseLuck or 0) / 10)
	local total = 0
	for i = 1, amount / 10 do
		local idx = level + i
		if idx > MAX_LUCK_LEVEL then
			return nil
		end
		total = total + (LEVEL_PRICES[idx] or 0)
	end
	return total
end

function tryLuckUpgrade()
	if not State.autoLuck then
		return false, "disabled"
	end
	if math.floor((State.baseLuck or 0) / 10) >= MAX_LUCK_LEVEL then
		return false, "max luck"
	end
	local now = tick()
	if now - (State.lastLuckUpgrade or 0) < LUCK_UPGRADE_COOLDOWN then
		return false, "cooldown"
	end
	local money = getLocalMoney()
	for _, amount in ipairs({ 100, 50, 10 }) do
		local cost = getLuckUpgradeCost(amount, State.baseLuck)
		if cost and money >= cost then
			State.lastLuckUpgrade = now
			REMOTES.PurchaseLuckUpgrade:FireServer(tostring(amount))
			return true, "+" .. amount .. "% luck"
		end
	end
	return false, "cannot afford luck"
end

function fireSeatPrompts(seat)
	if not (seat and seat.Parent) then
		return
	end
	local fired = {}
	local function tryPrompt(inst)
		if inst and inst.Parent and inst:IsA("ProximityPrompt") and not fired[inst] and inst.Enabled then
			fired[inst] = true
			if typeof(fireproximityprompt) == "function" then
				pcall(fireproximityprompt, inst, 0)
			end
		end
	end
	for _, desc in ipairs(seat:GetDescendants()) do
		tryPrompt(desc)
	end
	local chair = seat.Parent
	if chair then
		for _, desc in ipairs(chair:GetDescendants()) do
			tryPrompt(desc)
		end
		local tableModel = chair.Parent
		if tableModel then
			for _, desc in ipairs(tableModel:GetDescendants()) do
				if desc:IsA("ProximityPrompt") then
					local nearSeat = desc:FindFirstAncestorWhichIsA("Seat", true)
					if nearSeat == seat then
						tryPrompt(desc)
					end
				end
			end
		end
	end
end

function getSeatSitOffsets(seat)
	local y = seat and seat.Size and seat.Size.Y or 1
	local half = y * 0.5
	local list = {}
	for _, offset in ipairs(SIT_TRY_OFFSETS) do
		table.insert(list, offset)
	end
	table.insert(list, CFrame.new(0, half + 0.35, 0))
	table.insert(list, CFrame.new(0, half + 0.8, 0))
	table.insert(list, CFrame.new(0, half + 1.2, 0))
	return list
end

function attemptSitOnSeat(humanoid, root, seat)
	if not (humanoid and root and seat and seat.Parent) then
		return false
	end
	if seat.Occupant and seat.Occupant ~= humanoid then
		return false
	end
	if humanoid.SeatPart == seat then
		return true
	end

	configureHumanoidForSit(humanoid)
	pcall(function()
		seat.Disabled = false
	end)

	local savedHip = humanoid.HipHeight
	humanoid.HipHeight = math.min(savedHip, 1.1)
	fireSeatPrompts(seat)

	for _, offset in ipairs(getSeatSitOffsets(seat)) do
		local targetCf = seat.CFrame * offset
		root.CFrame = CFrame.lookAt(targetCf.Position, seat.Position)
		root.AssemblyLinearVelocity = Vector3.zero
		root.AssemblyAngularVelocity = Vector3.zero

		pcall(function()
			seat:Sit(humanoid)
		end)
		pcall(function()
			humanoid.Sit = true
		end)

		for _ = 1, 4 do
			RunService.Heartbeat:Wait()
			if humanoid.SeatPart == seat then
				humanoid.HipHeight = savedHip
				return true
			end
		end
	end

	pcall(function()
		root.CFrame = CFrame.lookAt((seat.CFrame * CFrame.new(0, 1.2, 0)).Position, seat.Position)
		seat:Sit(humanoid)
		humanoid.Sit = true
	end)
	for _ = 1, 8 do
		RunService.Heartbeat:Wait()
		if humanoid.SeatPart == seat then
			humanoid.HipHeight = savedHip
			return true
		end
	end

	humanoid.HipHeight = savedHip
	fireSeatPrompts(seat)
	return humanoid.SeatPart == seat
end

function distanceToSeat(root, seat)
	return (root.Position - seat.Position).Magnitude
end

function snapNearSeat(root, seat)
	root.CFrame = seat.CFrame * CFrame.new(0, 1.5, 0)
	root.AssemblyLinearVelocity = Vector3.zero
	root.AssemblyAngularVelocity = Vector3.zero
end

function stepTeleportToSeat(root, seat)
	local seatPos = (seat.CFrame * CFrame.new(0, 1.5, 0)).Position
	local delta = seatPos - root.Position
	local dist = delta.Magnitude
	if dist <= JOIN_SIT_RANGE then
		snapNearSeat(root, seat)
		return true
	end
	local move = math.min(dist - JOIN_SIT_RANGE + 1, JOIN_TELEPORT_STEP)
	if move <= 0 then
		snapNearSeat(root, seat)
		return true
	end
	root.CFrame = CFrame.new(root.Position + delta.Unit * move)
	root.AssemblyLinearVelocity = Vector3.zero
	root.AssemblyAngularVelocity = Vector3.zero
	return distanceToSeat(root, seat) <= JOIN_SIT_RANGE
end

function joinSeatPhysical(seat, tableName, chairName)
	local character = LocalPlayer.Character
	local humanoid = character and character:FindFirstChildOfClass("Humanoid")
	local root = character and character:FindFirstChild("HumanoidRootPart")
	if not (humanoid and root and seat and seat.Parent) then
		return false
	end

	if not tableName or not chairName then
		tableName, chairName = getTableFromSeat(seat)
	end

	if humanoid.SeatPart == seat then
		if tableName and chairName then
			confirmTableEnter(tableName, chairName)
		end
		return true
	end

	prepareForTableSit()
	if not ensureStanding(humanoid) then
		return false
	end

	if seat.Occupant and seat.Occupant ~= humanoid then
		return false
	end

	for _ = 1, 12 do
		if distanceToSeat(root, seat) <= JOIN_SIT_RANGE then
			break
		end
		if stepTeleportToSeat(root, seat) then
			break
		end
		task.wait(0.03)
	end
	snapNearSeat(root, seat)
	task.wait(0.08)

	for attempt = 1, 45 do
		if not (seat and seat.Parent) then
			return false
		end
		if seat.Occupant and seat.Occupant ~= humanoid then
			return false
		end

		attemptSitOnSeat(humanoid, root, seat)
		RunService.Heartbeat:Wait()
		RunService.Heartbeat:Wait()

		if humanoid.SeatPart == seat then
			if tableName and chairName then
				confirmTableEnter(tableName, chairName)
			end
			return true
		end

		if tableName and chairName and waitForTableJoin(tableName, chairName, 0.25) then
			return true
		end

		if attempt % 4 == 0 then
			snapNearSeat(root, seat)
			fireSeatPrompts(seat)
		end
		if attempt % 10 == 0 then
			prepareForTableSit()
			ensureStanding(humanoid)
			snapNearSeat(root, seat)
		end
	end

	if tableName and chairName and waitForTableJoin(tableName, chairName, 1) then
		return true
	end
	return humanoid.SeatPart == seat
end

function joinTableChair(tableModel, chairName, seat)
	if not tableModel or not chairName then
		return false
	end
	local tableName = tableModel.Name
	seat = seat or getSeatForChair(tableModel, chairName)

	local seated, tName, cName = getPlayerTableStatus(LocalPlayer)
	if seated and tName == tableName and cName == chairName then
		return true
	end

	prepareForTableSit()

	-- Game requires physical sit first; Watcher fires EnterRequest when SeatPart changes.
	if seat and seat.Parent then
		if joinSeatPhysical(seat, tableName, chairName) then
			return true
		end
	end

	return requestTableEnter(tableName, chairName, 3, 0.5)
end

function joinSeat(seat)
	local tableModel, chairName = getTableFromSeat(seat)
	if tableModel and chairName then
		return joinTableChair(tableModel, chairName, seat)
	end
	return joinSeatPhysical(seat)
end

function tryJoinPlayer(target)
	if not target then
		return false, "no player selected"
	end
	if target == LocalPlayer then
		return false, "cannot join yourself"
	end

	local atTable, tableName, chairName = getPlayerTableStatus(target)
	if not atTable then
		return false, target.DisplayName .. " is NOT at a table"
	end

	local character = target.Character
	local humanoid = character and character:FindFirstChildOfClass("Humanoid")
	local targetSeat = humanoid and humanoid.SeatPart
	local tableModel = getTableFromSeat(targetSeat)
	if not tableModel then
		return false, "invalid table"
	end

	local openSeat, openChairName = getOtherSeat(tableModel, chairName)
	if not openSeat or not openChairName then
		return false, tableName .. " is full"
	end

	local ok = joinTableChair(tableModel, openChairName, openSeat)
	return ok, ok and ("joined " .. tableName) or "failed to join table"
end

function queueJoinPlayer(target, onDone)
	if not target or State.isJoining or State.isForceLeaving then
		return false
	end
	if isLocalSeated() then
		return false
	end
	local now = tick()
	if now - State.lastJoinAttempt < JOIN_RETRY_COOLDOWN then
		return false
	end
	if not canJoinPlayer(target) then
		return false
	end

	State.isJoining = true
	State.lastJoinAttempt = now

	task.spawn(function()
		local ok, msg = false, "join failed"
		local callOk, err = pcall(function()
			ok, msg = tryJoinWithRetries(function()
				return tryJoinPlayer(target)
			end)
		end)
		State.isJoining = false
		if not callOk then
			warn("[Dutware] join error:", err)
			msg = tostring(err)
		elseif ok then
			State.lastJoin = tick()
			setStatus("Joined — " .. tostring(msg))
		end
		if onDone then
			onDone(ok, msg)
		end
	end)
	return true
end

	return remote
end

function findQuickChatButtonText(matchFn)
	local playerGui = LocalPlayer:FindFirstChild("PlayerGui")
	local quickChatGui = playerGui and playerGui:FindFirstChild("QuickChatGUI")
	local panel = quickChatGui and quickChatGui:FindFirstChild("QuickChat")
	local buttonsList = panel and panel:FindFirstChild("ButtonsList")
	if not buttonsList then
		return nil
	end
	for _, btn in ipairs(buttonsList:GetChildren()) do
		if btn:IsA("ImageButton") then
			local label = btn:FindFirstChildOfClass("TextLabel")
			local text = label and label.Text or ""
			if text ~= "" and matchFn(text) then
				return text
			end
		end
	end
	return nil
end

function resolveQuickChatMessage(preferred, matchFn)
	return findQuickChatButtonText(matchFn) or preferred
end

function isQuickChatOpen()
	if not isLocalSeated() then
		return false
	end
	if State.matchActive then
		return true
	end
	local playerGui = LocalPlayer:FindFirstChild("PlayerGui")
	local quickChatGui = playerGui and playerGui:FindFirstChild("QuickChatGUI")
	local panel = quickChatGui and quickChatGui:FindFirstChild("QuickChat")
	if not panel then
		return false
	end
	return panel.Position.X.Scale < 1.15
end

function sendQuickChat(message)
	local remote = getQuickChatRemote()
	if not remote or not message or message == "" then
		return false
	end
	remote:FireServer(message)
	return true
end

function tryAutoQuickChat(now)
	if not State.matchActive or not isLocalSeated() then
		return
	end
	if State.autoQuickChatLetsSplit and now - (tryAutoQuickChat.lastSplit or 0) >= QUICK_CHAT_INTERVAL then
		tryAutoQuickChat.lastSplit = now
		local splitText = resolveQuickChatMessage(QUICK_CHAT_LETS_SPLIT, function(text)
			return text:lower():find("split", 1, true) ~= nil
		end)
		sendQuickChat(splitText)
	end
end

function getAutoDecisionChoice()
	local choice = State.decision == "Steal" and "Steal" or "Split"
	if State.autoStealOnRarity and choice == "Split" and State.matchRollRarity then
		if isRarityInSet(State.stealRarities, State.matchRollRarity) then
			return "Steal"
		end
	end
	return choice
end

function fireDecision(choice)
	choice = choice or State.decision
	if choice ~= "Split" and choice ~= "Steal" then
		return false
	end
	REMOTES.DecisionRequest:FireServer(choice)
	setStatus("DecisionRequest -> " .. choice)
	return true
end

function cancelAutoDecision()
	decisionToken = decisionToken + 1
	State.inDecision = false
end

function submitAutoDecision()
	if not State.autoDecision then
		return
	end
	decisionToken = decisionToken + 1
	local token = decisionToken
	State.inDecision = true

	task.spawn(function()
		local started = tick()
		while token == decisionToken and tick() - started < 28 do
			if not isLocalSeated() then
				State.inDecision = false
				return
			end

			if isDecisionOpen() then
				local choice = getAutoDecisionChoice()
				if choice == "Steal" and State.decision ~= "Steal" then
					setStatus("Auto steal — matched rarity: " .. tostring(State.matchRollRarity))
				end
				fireDecision(choice)
				State.inDecision = false
				return
			end
			task.wait(0.02)
		end
		State.inDecision = false
	end)
end

function switchTab(tabIndex)
	if activeTab == tabIndex or not UI.tab1Panel or not UI.tab2Panel or not UI.tab3Panel or not UI.tab4Panel or not UI.tab5Panel then
		return
	end
	if dropdownOpen then
		setDropdownOpen(false)
	end
	if UI.sellRarityDropdownOpen then
		setSellRarityDropdownOpen(false)
	end
	if UI.webhookRarityDropdownOpen then
		setWebhookRarityDropdownOpen(false)
	end
	if UI.stealRarityDropdownOpen then
		setStealRarityDropdownOpen(false)
	end
	if UI.merchantSlotDropdownOpen then
		setMerchantSlotDropdownOpen(false)
	end
	if UI.traitStopDropdownOpen then
		setTraitStopDropdownOpen(false)
	end
	if UI.chairPickDropdownOpen then
		setChairPickDropdownOpen(false)
	end

	local prev = activeTab
	activeTab = tabIndex
	local tabCount = 5
	local slideInfo = TweenInfo.new(0.38, Enum.EasingStyle.Quint, Enum.EasingDirection.Out)
	local fadeInfo = TweenInfo.new(0.28, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)

	if UI.tabIndicator then
		tween(UI.tabIndicator, slideInfo, {
			Position = UDim2.new((tabIndex - 1) / tabCount, 2, 0, 2),
		}):Play()
	end

	local panels = { UI.tab1Panel, UI.tab2Panel, UI.tab3Panel, UI.tab4Panel, UI.tab5Panel }
	local buttons = { UI.tabBtnTable, UI.tabBtnProfile, UI.tabBtnSell, UI.tabBtnTrait, UI.tabBtnMisc }
	local outPanel = panels[prev]
	local inPanel = panels[tabIndex]

	for i, panel in ipairs(panels) do
		if i ~= tabIndex and i ~= prev then
			panel.Visible = false
		end
	end

	outPanel.Visible = true
	inPanel.Visible = true

	tween(outPanel, fadeInfo, {
		Position = UDim2.new(prev < tabIndex and -1 or 1, 0, 0, 0),
	}):Play()
	inPanel.Position = UDim2.new(tabIndex > prev and 1 or -1, 0, 0, 0)
	tween(inPanel, slideInfo, {
		Position = UDim2.new(0, 0, 0, 0),
	}):Play()

	task.delay(0.32, function()
		if activeTab ~= tabIndex then
			return
		end
		outPanel.Visible = false
	end)

	local activeColor = Color3.fromRGB(235, 240, 255)
	local idleColor = Color3.fromRGB(130, 140, 165)
	for i, btn in ipairs(buttons) do
		if btn then
			tween(btn, TweenInfo.new(0.25), {
				TextColor3 = i == tabIndex and activeColor or idleColor,
			}):Play()
		end
	end

	if tabIndex == 3 then
		updateSellLabels()
	end
	if tabIndex == 4 then
		updateTraitTabLabels()
	end
	if tabIndex == 5 then
		updateMiscSectionLayout()
	end
	if tabIndex == 5 then
		updateWebhookLabels()
	end
	resizeHubLayout(dropdownOpen)
end

function applyMerchantTimerSync(restockId, _, purchases)
	local changed = restockId ~= State.merchantRestockId
	if changed then
		syncMerchantStockFromRestock(restockId)
	else
		State.merchantRestockId = restockId or State.merchantRestockId
	end
	if type(purchases) == "table" then
		if changed then
			State.merchantPurchasedThisCycle = {}
		end
		for slotId, count in pairs(purchases) do
			State.merchantPurchasedThisCycle[slotId] = math.max(State.merchantPurchasedThisCycle[slotId] or 0, count)
		end
	end
end

function rebuildDropdown()
	for _, option in ipairs(UI.dropdownOptions) do
		if option.frame then
			option.frame:Destroy()
		end
	end
	table.clear(UI.dropdownOptions)

	if not UI.dropdownList then
		return
	end

	local sorted = getPlayersSortedByLuck()
	local rankMap = {}
	for i, entry in ipairs(sorted) do
		rankMap[entry.player.UserId] = i
	end

	for _, player in ipairs(Players:GetPlayers()) do
		local atTable, tableName = getPlayerTableStatus(player)
		local luck = getLuck(player)
		local tag = player == LocalPlayer and " (you)" or ""
		local rank = rankMap[player.UserId]
		local rankText = rank and ("#" .. rank .. " ") or ""
		local tableText = atTable and ("IN " .. tableName) or "NOT IN TABLE"

		local row = Instance.new("TextButton")
		row.Size = UDim2.new(1, -4, 0, 36)
		row.BackgroundColor3 = Color3.fromRGB(42, 45, 54)
		row.BorderSizePixel = 0
		row.AutoButtonColor = false
		row.Text = ""
		row.Parent = UI.dropdownList

		Instance.new("UICorner", row).CornerRadius = UDim.new(0, 6)

		local thumb = Instance.new("ImageLabel", row)
		thumb.BackgroundColor3 = Color3.fromRGB(30, 32, 40)
		thumb.Position = UDim2.new(0, 4, 0.5, -14)
		thumb.Size = UDim2.new(0, 28, 0, 28)
		thumb.Image = getAvatarUrl(player.UserId)
		thumb.ScaleType = Enum.ScaleType.Crop
		Instance.new("UICorner", thumb).CornerRadius = UDim.new(1, 0)

		local nameLabel = Instance.new("TextLabel", row)
		nameLabel.BackgroundTransparency = 1
		nameLabel.Position = UDim2.new(0, 38, 0, 0)
		nameLabel.Size = UDim2.new(1, -42, 1, 0)
		nameLabel.Font = Enum.Font.Gotham
		nameLabel.TextSize = 11
		nameLabel.TextXAlignment = Enum.TextXAlignment.Left
		nameLabel.TextColor3 = atTable and Color3.fromRGB(120, 255, 160) or Color3.fromRGB(220, 220, 220)
		nameLabel.TextTruncate = Enum.TextTruncate.AtEnd
		nameLabel.Text = string.format("%s%s%s | %s | %s", rankText, player.DisplayName, tag, formatLuckText(player), tableText)

		row.MouseEnter:Connect(function()
			tween(row, TweenInfo.new(0.15), { BackgroundColor3 = Color3.fromRGB(58, 62, 78) }):Play()
		end)
		row.MouseLeave:Connect(function()
			tween(row, TweenInfo.new(0.15), { BackgroundColor3 = Color3.fromRGB(42, 45, 54) }):Play()
		end)
		row.MouseButton1Click:Connect(function()
			selectPlayer(player)
		end)

		table.insert(UI.dropdownOptions, { player = player, frame = row })
	end

	local layout = UI.dropdownList:FindFirstChildOfClass("UIListLayout")
	if layout then
		UI.dropdownList.CanvasSize = UDim2.new(0, 0, 0, layout.AbsoluteContentSize.Y + 6)
	end
end

local REMOTE_BIND_TIMEOUT = 8

function bindRemotes()
	local function connectPlayerEvent(name, handler)
		local event = PlayerEvents:WaitForChild(name, REMOTE_BIND_TIMEOUT)
		if event then
			event.OnClientEvent:Connect(handler)
		else
			warn("[Dutware] Missing Player event:", name)
		end
	end

	REMOTES.RequestPlayerData:FireServer()

	connectPlayerEvent("MoneyUpdated", function(amount)
		State.money = amount or 0
		updateMoneyLabel()
		if State.autoLuck then
			tryLuckUpgrade()
		end
	end)

	connectPlayerEvent("BaseLuckUpdated", function(luck)
		State.baseLuck = luck or 0
		updateMoneyLabel()
		if State.autoLuck then
			tryLuckUpgrade()
		end
	end)

	connectPlayerEvent("LuckUpdated", function(luck)
		State.playerLuck[LocalPlayer.UserId] = luck or 0
	end)

	connectPlayerEvent("ChairShopUpdated", function(owned, equipped, money)
		State.ownedChairs = normalizeOwnedChairs(owned)
		State.equippedChair = equipped or "DefaultChair"
		State.money = money or State.money
		table.clear(chairLuckCache)
		updateMoneyLabel()
		if UI.chairPickList then
			rebuildChairPickDropdown()
		end
		updateTraitTabLabels()
	end)

	local specialChairUpdated = PlayerEvents:FindFirstChild("SpecialChairUpdated")
	if specialChairUpdated then
		specialChairUpdated.OnClientEvent:Connect(function(payload)
			if type(payload) == "table" and type(payload.ownedChairs) == "table" then
				State.specialOwnedChairs = normalizeOwnedChairs(payload.ownedChairs)
				table.clear(chairLuckCache)
				updateProfileLabels()
				if UI.chairPickList then
					rebuildChairPickDropdown()
				end
				updateTraitTabLabels()
			end
		end)
	end

	connectPlayerEvent("EquippedChairUpdated", function(chairName)
		State.equippedChair = chairName or State.equippedChair
		updateProfileLabels()
	end)

	REMOTES.InventoryUpdated.OnClientEvent:Connect(function(inventory, hotbar, favourites, _, _, cashMult)
		State.inventory = inventory or {}
		State.hotbarIds = hotbar or {}
		State.favouriteIds = favourites or {}
		State.inventoryMax = math.max(State.inventoryMax, #State.inventory)
		if cashMult then
			State.cashMult = math.max(1, cashMult)
		end
		updateProfileLabels()
		updateSellLabels()
	end)

	local luckBroadcast = PlayerEvents:FindFirstChild("LuckBroadcast")
	if luckBroadcast then
		luckBroadcast.OnClientEvent:Connect(function(userId, luck)
			State.playerLuck[userId] = luck or 0
		end)
	end

	local boostUpdated = PlayerEvents:FindFirstChild("BoostUpdated")
	if boostUpdated then
		boostUpdated.OnClientEvent:Connect(function(boostName, expiry)
			if LUCK_BOOST_MULT[boostName] then
				if expiry == 0 or not expiry then
					State.luckBoosts[boostName] = nil
				else
					State.luckBoosts[boostName] = expiry
				end
				refreshLuckEventMult()
				updateProfileLabels()
			end
		end)
	end

	task.spawn(function()
		while true do
			refreshLuckEventMult()
			task.wait(1)
		end
	end)

	local leaderstats = LocalPlayer:WaitForChild("leaderstats", 15)
	if leaderstats then
		local moneyValue = leaderstats:FindFirstChild("Money")
		if moneyValue then
			State.money = moneyValue.Value
			moneyValue.Changed:Connect(function(value)
				State.money = value
				updateMoneyLabel()
			end)
		end
	end

	REMOTES.MatchStarted.OnClientEvent:Connect(function(tableName, _, _, rollName, rollVariant, rollRarity)
		if isLocalSeated() then
			State.localRollActive = true
			State.matchActive = true
			if isLocalPlayerAtTable(tableName) then
				State.matchRollName = rollName
				State.matchRollVariant = rollVariant
				State.matchRollRarity = normalizeRollRarity(rollRarity) or rollRarity
			end
		end
		if isLocalPlayerAtTable(tableName) then
			setStatus("Match started — auto decision queued")
			submitAutoDecision()
			if rollName and rollName ~= "" then
				onRollPrediction(tableName, rollName, rollVariant, rollRarity)
			end
		else
			setStatus("Match started")
		end
	end)

	REMOTES.MatchEnded.OnClientEvent:Connect(function()
		cancelAutoDecision()
		State.localRollActive = false
		State.matchActive = false
		State.matchRollName = nil
		State.matchRollVariant = nil
		State.matchRollRarity = nil
	end)

	REMOTES.MatchResolved.OnClientEvent:Connect(function(
		tableName,
		_,
		_,
		result,
		brainrotName,
		variant,
		opponentName
	)
		cancelAutoDecision()
		State.localRollActive = false
		State.matchActive = false

		if isLocalPlayerAtTable(tableName) and result == "win" and brainrotName and brainrotName ~= "" then
			local rarity = State.matchRollRarity or getBrainrotRarity(brainrotName, nil)
			task.spawn(function()
				tryWebhookOnWin(brainrotName, variant, rarity, opponentName)
				tryWebhookOnSummerPet(brainrotName, variant, opponentName)
			end)
		end

		State.matchRollName = nil
		State.matchRollVariant = nil
		State.matchRollRarity = nil
	end)

	REMOTES.RollReveal.OnClientEvent:Connect(function(tableName, name, variant, rarity)
		if isLocalPlayerAtTable(tableName) then
			State.matchRollName = name
			State.matchRollVariant = variant
			State.matchRollRarity = normalizeRollRarity(rarity) or rarity
		end
		onRollPrediction(tableName, name, variant, rarity)
	end)

	if REMOTES.RollStarted then
		REMOTES.RollStarted.OnClientEvent:Connect(function(tableName, _, _, _, _, tableLuck, uid1, uid2)
			local id = LocalPlayer.UserId
			local inMatch = isLocalPlayerAtTable(tableName)
			if uid1 or uid2 then
				inMatch = inMatch or (id == uid1 or id == uid2)
			end
			if not inMatch then
				return
			end
			if type(tableLuck) == "number" then
				State.tableLuck = tableLuck
			end
			State.localRollActive = true
		end)
	end

	if REMOTES.SpinSequenceUpdate then
		REMOTES.SpinSequenceUpdate.OnClientEvent:Connect(function(sequence)
			if type(sequence) == "table" then
				State.spinSequence = sequence
			end
		end)
	end

	if REMOTES.MultiplierInfo then
		REMOTES.MultiplierInfo.OnClientEvent:Connect(function(mult)
			State.tableMultiplier = mult or 1
		end)
	end

	connectPlayerEvent("RebirthUpdated", function(rebirthCount, cashMult, money, cost)
		State.rebirthCount = rebirthCount or 0
		if cashMult then
			State.cashMult = math.max(1, cashMult)
		end
		if money ~= nil then
			State.money = money
		end
		State.rebirthCost = cost or getRebirthCost(State.rebirthCount)
		updateProfileLabels()
	end)

	local cashBoostUpdated = PlayerEvents:FindFirstChild("CashBoostUpdated")
	if cashBoostUpdated then
		cashBoostUpdated.OnClientEvent:Connect(function(owned)
			State.vipDouble = owned == true
		end)
	end

	task.spawn(function()
		local ok, owns = pcall(function()
			return MarketplaceService:UserOwnsGamePassAsync(LocalPlayer.UserId, VIP_GAMEPASS_ID)
		end)
		if ok then
			State.vipDouble = owns == true
		end
	end)

	if REMOTES.MerchantTimerSync then
		REMOTES.MerchantTimerSync.OnClientEvent:Connect(function(restockId, restockTime, purchases)
			applyMerchantTimerSync(restockId, restockTime, purchases)
		end)
		syncMerchantStockFromRestock(0)
	end

	if REMOTES.MerchantBuyResult then
		REMOTES.MerchantBuyResult.OnClientEvent:Connect(function(slotId, success, _, _, reason)
			if slotId then
				State.merchantPendingBuy[slotId] = nil
			end
			if success and slotId then
				State.merchantPurchasedThisCycle[slotId] = (State.merchantPurchasedThisCycle[slotId] or 0) + 1
			elseif not success and reason == "no_cash" and State.autoBuyMerchant then
				setStatus("Merchant: not enough cash for " .. tostring(slotId))
			end
		end)
	end

	connectPlayerEvent("TraitRerollsUpdated", function(count)
		State.traitRerolls = tonumber(count) or 0
		updateTraitTabLabels()
	end)

	if REMOTES.ChairTraitUpdated then
		REMOTES.ChairTraitUpdated.OnClientEvent:Connect(handleChairTraitUpdated)
	end

	if REMOTES.IndexSlotsUpdated then
		REMOTES.IndexSlotsUpdated.OnClientEvent:Connect(function(cap, pending, unique)
			State.inventoryCap = cap or State.inventoryCap
			State.indexSlotsPending = pending or 0
			State.indexUniqueCount = unique or 0
			_G.InventoryCap = State.inventoryCap
			_G.IndexSlotsPending = State.indexSlotsPending
			_G.IndexUniqueCount = State.indexUniqueCount
			updateProfileLabels()
		end)
	end

	task.defer(function()
		syncIndexFromGame()
	end)

	task.defer(refreshChairTraitsFromRemote)

	task.defer(function()
		State.indexSlotsPending = _G.IndexSlotsPending or State.indexSlotsPending
		State.indexUniqueCount = _G.IndexUniqueCount or State.indexUniqueCount
		State.inventoryCap = _G.InventoryCap or State.inventoryCap
		updateProfileLabels()
	end)

	task.delay(2, function()
		REMOTES.RequestPlayerData:FireServer()
		REMOTES.RequestInventory:FireServer()
	end)
end

	knob.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
	knob.BorderSizePixel = 0
	Instance.new("UICorner", knob).CornerRadius = UDim.new(1, 0)

	local enabled = config.default == true
	local onColor = config.onColor or Color3.fromRGB(80, 200, 120)
	local offColor = Color3.fromRGB(55, 58, 68)

	local function setState(on, silent)
		enabled = on == true
		tween(track, TweenInfo.new(0.2, Enum.EasingStyle.Quad), {
			BackgroundColor3 = enabled and onColor or offColor,
		}):Play()
		tween(knob, TweenInfo.new(0.22, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
			Position = UDim2.new(0, enabled and 24 or 2, 0.5, -10),
		}):Play()
		if config.onChange and not silent then
			config.onChange(enabled)
		end
	end

	row.MouseButton1Click:Connect(function()
		setState(not enabled)
	end)
	setState(enabled, true)

	return {
		frame = row,
		label = label,
		track = track,
		knob = knob,
		set = setState,
		get = function()
			return enabled
		end,
	}
end

function resetHubGuiState()
	if State.fpsBoostConn then
		State.fpsBoostConn:Disconnect()
		State.fpsBoostConn = nil
	end
	if State.antiAfkConn then
		State.antiAfkConn:Disconnect()
		State.antiAfkConn = nil
	end
	for _, conn in ipairs(State.hidePlayerConns) do
		conn:Disconnect()
	end
	table.clear(State.hidePlayerConns)

	local old = getGuiParent():FindFirstChild("DutwareTableHub")
	if old then
		old:Destroy()
	end
	local oldAfk = getGuiParent():FindFirstChild("DutwareAfkScreen")
	if oldAfk then
		oldAfk:Destroy()
	end
end

function resetJoinTimers()
	State.lastJoin = 0
	State.lastLeave = 0
	State.lastJoinAttempt = 0
end

RunService.Heartbeat:Connect(function()
	local now = tick()

	if State.joinCooldownReset then
		State.lastJoin = 0
		State.lastLeave = 0
		State.lastJoinAttempt = 0
		State.joinCooldownReset = false
	end

	if now - State.lastFarm >= 1.5 and not dropdownOpen then
		State.lastFarm = now
		if State.autoLuck then
			tryLuckUpgrade()
			updateProfileLabels()
		end
		if State.autoChair then
			tryChairUpgrade()
			updateProfileLabels()
		end
		if State.autoRebirth then
			tryAutoRebirth()
		end
		if State.autoIndexReward then
			tryAutoIndexReward()
		end
	end

	if State.autoCollect and now - State.lastCollect >= COLLECT_INTERVAL then
		State.lastCollect = now
		tryCollectAll()
	end

	tryHourlyProfileWebhook()

	if now - State.lastAutoSell >= SELL_COOLDOWN and State.autoSellRarity and fireQuickSell() then
		State.lastAutoSell = now
	end

	if now - State.lastAutoSellLowDps >= SELL_COOLDOWN and State.autoSellBelowMoney and trySellPetsBelowDps() then
		State.lastAutoSellLowDps = now
	end

	tryAutoQuickChat(now)

	tryMerchantAutoBuy()
	tryAutoTraitReroll()

	local shouldLeave = false
	local okLeave, leaveResult = pcall(shouldForceLeaveSeat)
	if okLeave then
		shouldLeave = leaveResult
	end

	if isLocalSeated() and shouldLeave then
		State.aloneAtTableSince = State.aloneAtTableSince or now
		if now - State.aloneAtTableSince >= 2.5 and now - State.lastLeave >= 0.35 and not State.isForceLeaving then
			State.lastLeave = now
			State.aloneAtTableSince = nil
			forceLeaveTable(function(left)
				if left then
					setStatus("Left table — waiting for next action")
				end
			end)
			return
		end
	elseif not isLocalSeated() or not shouldLeave then
		State.aloneAtTableSince = nil
	end

	if not hasAnyAutoJoinMode() then
		updateLuckWaitLabel(nil, 0, "off")
	elseif isLocalSeated() then
		if State.autoJoinHighestLuck then
			local top = getHighestLuckEntry()
			if top then
				local opponent = getTableOpponent()
				if opponent == top.player then
					updateLuckWaitLabel(top.player, top.luck, "joined")
				else
					updateLuckWaitLabel(top.player, top.luck, opponent and "wait" or "wait")
				end
			end
		end

		if State.autoJoinMinLuck then
			local opponent = getTableOpponent()
			local oppLuck = opponent and getLuck(opponent) or 0
			if opponent and oppLuck >= MIN_LUCK_JOIN then
				updateMinLuckWaitLabel(
					string.format("Playing with %s (%s)", opponent.DisplayName, formatLuckText(opponent)),
					Color3.fromRGB(100, 255, 150)
				)
			elseif opponent and oppLuck < MIN_LUCK_JOIN then
				updateMinLuckWaitLabel("In match — waiting to leave", Color3.fromRGB(180, 160, 255))
			else
				updateMinLuckWaitLabel("At table — waiting for partner", Color3.fromRGB(180, 160, 255))
			end
		end

		if State.autoJoinRandom and UI.tableStatusLabel then
			local opponent = getTableOpponent()
			if opponent then
				UI.tableStatusLabel.Text = "Table: random player — " .. opponent.DisplayName
				UI.tableStatusLabel.TextColor3 = Color3.fromRGB(100, 255, 150)
			else
				UI.tableStatusLabel.Text = "Table: random player — waiting for partner"
				UI.tableStatusLabel.TextColor3 = Color3.fromRGB(255, 200, 100)
			end
		elseif State.autoJoinRandomChair and UI.tableStatusLabel then
			local opponent = getTableOpponent()
			if opponent then
				UI.tableStatusLabel.Text = "Table: random table — " .. opponent.DisplayName
				UI.tableStatusLabel.TextColor3 = Color3.fromRGB(100, 255, 150)
			else
				UI.tableStatusLabel.Text = "Table: random table — waiting for partner"
				UI.tableStatusLabel.TextColor3 = Color3.fromRGB(255, 200, 100)
			end
		end
	elseif not State.isJoining then
		if State.autoJoin and State.selectedPlayer then
			queueJoinPlayer(State.selectedPlayer, function(ok, msg)
				if ok then
					setStatus("Joined selected user — " .. msg)
					if UI.tableStatusLabel then
						UI.tableStatusLabel.Text = "Table: " .. State.selectedPlayer.DisplayName
						UI.tableStatusLabel.TextColor3 = Color3.fromRGB(100, 255, 150)
					end
				end
			end)
		elseif State.autoJoinHighestLuck then
			local top = getHighestLuckEntry()
			if top then
				local target, luck, _, mode = findHighestLuckJoinTarget()
				if target then
					queueJoinPlayer(target, function(ok, msg)
						if ok then
							updateLuckWaitLabel(target, luck, "joined")
							setStatus(string.format("Joined %s (%d%%) — %s", target.DisplayName, luck, msg))
						end
					end)
				elseif mode == "wait" then
					updateLuckWaitLabel(top.player, top.luck, "wait")
					if now - State.lastJoin >= 2 then
						State.lastJoin = now
						setStatus(string.format("Waiting for %s (%d%%) to join", top.player.DisplayName, top.luck))
					end
				else
					updateLuckWaitLabel(top.player, top.luck, "wait")
				end
			end
		elseif State.autoJoinMinLuck then
			local target, luck = findMinLuckJoinTarget()
			if target then
				queueJoinPlayer(target, function(ok, msg)
					if ok then
						updateMinLuckWaitLabel(
							string.format("Joined %s (%s)", target.DisplayName, formatLuckText(target)),
							Color3.fromRGB(100, 255, 150)
						)
						setStatus(string.format("Joined %s (%d%%) — %s", target.DisplayName, luck, msg))
					end
				end)
			else
				updateMinLuckWaitLabel("Waiting for 1000%+ player at table", Color3.fromRGB(180, 160, 255))
			end
		elseif State.autoJoinRandom then
			local target, luck, mode = findRandomJoinTarget()
			if target and mode == "join" then
				queueJoinPlayer(target, function(ok, msg)
					if ok then
						setStatus(string.format("Joined random player %s (%d%%) — %s", target.DisplayName, luck, msg))
						if UI.tableStatusLabel then
							UI.tableStatusLabel.Text = "Table: joined " .. target.DisplayName
							UI.tableStatusLabel.TextColor3 = Color3.fromRGB(100, 255, 150)
						end
					end
				end)
			elseif now - State.lastJoin >= 2 then
				State.lastJoin = now
				setStatus("Waiting for random player at table...")
				if UI.tableStatusLabel then
					UI.tableStatusLabel.Text = "Table: searching random player..."
					UI.tableStatusLabel.TextColor3 = Color3.fromRGB(180, 160, 255)
				end
			end
		elseif State.autoJoinRandomChair then
			if queueJoinRandomChair(function(ok, msg)
				if ok and UI.tableStatusLabel then
					UI.tableStatusLabel.Text = "Table: joined random table"
					UI.tableStatusLabel.TextColor3 = Color3.fromRGB(100, 255, 150)
				end
			end) then
				return
			end
			if now - State.lastJoin >= 2 then
				State.lastJoin = now
				local openCount = #getJoinableRandomTableSeats()
				setStatus("Waiting for open table... (" .. openCount .. " found)")
				if UI.tableStatusLabel then
					UI.tableStatusLabel.Text = "Table: searching open table..."
					UI.tableStatusLabel.TextColor3 = Color3.fromRGB(255, 180, 100)
				end
			end
		end
	end
end)
